% In this script we investigate the timing as a function of the numer of
% matvecs as the system size increases
clear;%close all;

delta = 1/300;
rho   = 1/20;

nscale  = 2;
scale0  = 13;
tol     = 1e-3;

nrea = 2;
iterations0    = 50;
iterations1    = 100;
iterations     = floor(linspace(iterations0,iterations1,nrea));

errorspg=zeros(nrea,nscale,5);
erroramp=zeros(nrea,nscale,5);

if 1,
    for iscale=1:nscale
        N = 2^(scale0+iscale);
        n = floor(delta*N);
        k = floor(rho*n);
        inds=randperm(N);
        ind=inds(1:n);
        R=opRestriction(N,ind);
        A=1/sqrt(n)*R*sqrt(N)*opRomberg([N]);
        x = [sign(rand(k,1) - 0.5); zeros(N-k,1)];
        x = x(randperm(N));
        tau0=norm(x,1);
        % Generate Measurements
        b = A*x;
        for irea=1:nrea
            % Do the one-norm solve
            niterations = iterations(irea);
            opt.verbosity=0;
            opt.iterations=niterations;
            %[xhatspg,r,~,infospg{irea,iscale}] = spg_bpdn(A, b, tol*norm(b,2),opt);
            [xhatspg,r,~,infospg{irea,iscale}] = spgl1(A, b, 0, 1e-3, zeros(size(A,2),1), opt);
            errorspg(irea,iscale,1)=N;
            errorspg(irea,iscale,2)=infospg{irea,iscale}.nProdA+infospg{irea,iscale}.nProdAt;
            errorspg(irea,iscale,3)=norm(r,2);
            errorspg(irea,iscale,4)=norm(x-xhatspg,2);
            errorspg(irea,iscale,5)=norm(x,2);

            [xhatpqn,r,~,infopqn{irea,iscale}] = pqnl1(A, b, 0, 1e-3, zeros(size(A,2),1), opt);
            errorpqn(irea,iscale,1)=N;
            errorpqn(irea,iscale,2)=infopqn{irea,iscale}.nProdA+infopqn{irea,iscale}.nProdAt;
            errorpqn(irea,iscale,3)=norm(r,2);
            errorpqn(irea,iscale,4)=norm(x-xhatspg,2);
            errorpqn(irea,iscale,5)=norm(x,2);

            [xhatamp,infoamp{irea,iscale}] = reconstructAmpcountfast(A, b, niterations, tol);
            erroramp(irea,iscale,1)=N;
            erroramp(irea,iscale,2)=infoamp{irea,iscale}.nProdAAt;
            erroramp(irea,iscale,3)=norm(A*xhatamp-b,2);
            erroramp(irea,iscale,4)=norm(x-xhatamp,2);
            erroramp(irea,iscale,5)=norm(x,2);

        end
    end

    save varyingiterations errorspg errorpqn erroramp N delta rho iterations infospg infopqn infoamp
else
    load varyingiterations
end
